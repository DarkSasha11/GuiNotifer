task.spawn(function()
    while task.wait(1.5) do
        local success, response_body = pcall(game.HttpGet, game, API_URL, true)
        if not success then continue end
        
        local decoded_success, api_data = pcall(Http.JSONDecode, Http, response_body)
        if not decoded_success or type(api_data) ~= "table" then continue end
        
        for _, item in ipairs(api_data) do
            local id_str = tostring(item and item.id)
            if id_str and not PROCESSED_LOG_IDS[id_str] then
                PROCESSED_LOG_IDS[id_str] = item 
                local card = createLogCard(item)
                card.Visible = shouldShowLog(item)
                
                if card.Visible then sound:Play() end
                
                Debris:AddItem(card, LOG_LIFETIME) 
                
                if AUTO_JOIN_ENABLED and card.Visible then TPS:TeleportToPlaceInstance(tonumber(item.placeId), item.jobId, Players.LocalPlayer); task.wait(5) end
            end
        end
        -- Ð£Ð´Ð°Ð»ÐµÐ½Ð¸Ðµ ÑÑ‚Ð°Ñ€Ñ‹Ñ… ID Ð¸Ð· Ð¿Ð°Ð¼ÑÑ‚Ð¸
        for id, data in pairs(PROCESSED_LOG_IDS) do
            if os.time() - (data.createdAt_unix or 0) > (LOG_LIFETIME + 5) then PROCESSED_LOG_IDS[id] = nil end
        end
    end
end)
